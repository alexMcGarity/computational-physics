#version of the given code that scatterplots all location data and generates a circle

import math
import random as rnd
import matplotlib.pyplot as plt

pi = 4.0 * math.atan2(1, 1)

points = int(input("Enter number of MC points per trial [500] : ") or 500)
trials = int(input("Enter number of trials [20] : ") or 20)
rnd.seed(int(input("Enter random number seed [ 0 => system ] : ") or 0) or None)

ratio = []

for i in range(trials):
    bounded = 0
    for j in range(points):
        x = rnd.uniform(-1.0, +1.0)
        y = rnd.uniform(-1.0, +1.0)
        bounded += (math.sqrt(x**2 + y**2)<= 1)
        if math.sqrt(x**2 + y**2) <= 1:
            plt.scatter(x, y, color="0.0")
        else:
            plt.scatter(x, y, color="0.5")
    ratio.append(4.0 * bounded / points / pi)

plt.title("Monte Carlo Simulation of a Circle")
plt.xlabel("x")
plt.ylabel("y")
plt.grid()
plt.show()
